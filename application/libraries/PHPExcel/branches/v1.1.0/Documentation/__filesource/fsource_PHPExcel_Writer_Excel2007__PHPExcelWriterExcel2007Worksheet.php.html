  52.     public function writeWorksheet($pSheet = null, $pStringTable = null)
  53.     {
  54.         if (!is_null($pSheet)) {
  55.             // Create XML writer
  56.             $objWriter = new xmlWriter();

  58.             
  59.             // XML header

  61.

  67.             
  68.                 // sheetPr
  69.                 $this->_writeSheetPr($objWriter, $pSheet);
  70.             
  71.                 // Dimension
  72.                 $this->_writeDimension($objWriter, $pSheet);
  73.                 
  74.                 // sheetViews
  75.                 $this->_writeSheetViews($objWriter);
  76.             
  77.                 // sheetFormatPr
  78.                 $this->_writeSheetFormatPr($objWriter);
  79.                 
  80.                 // cols
  81.                 $this->_writeCols($objWriter, $pSheet);
  82.                         
  83.                 // sheetData
  84.                 $this->_writeSheetData($objWriter, $pSheet, $pStringTable);
  85.                 
  86.                 // sheetProtection
  87.                 $this->_writeSheetProtection($objWriter);
  88.                 
  89.                 // conditionalFormatting

  91.                 
  92.                 // Print options
  93.                 $this->_writePrintOptions($objWriter);
  94.                                 
  95.                 // Page margins
  96.                 $this->_writePageMargins($objWriter, $pSheet);
  97.

  99.                 $this->_writePageSetup($objWriter, $pSheet);

  109.                 
  110.             $objWriter->endElement();
  111.  
  112.             // Return

  211.     {
  212.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet) {
  213.             // cols
  214.             $objWriter->startElement('cols');
  215.                     
  216.                 // Check if there is at least one column dimension specified. If not, create one.
  217.                 if (count($pSheet->getColumnDimensions()) == 0) {
  218.                     $pSheet->getColumnDimension('A')->setWidth(10);
  219.                 }
  220.                     
  221.                 // Loop trough column dimensions
  222.                 foreach ($pSheet->getColumnDimensions() as $colDimension) {
  223.                     // col
  224.                     $objWriter->startElement('col');
  225.                     $objWriter->writeAttribute('min',     PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
  226.                     $objWriter->writeAttribute('max',     PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
  227.                         
  228.                     if ($colDimension->getWidth() < 0) {
  229.                         $objWriter->writeAttribute('bestFit',         '1');
  230.                         $objWriter->writeAttribute('width',         '10');
  231.                     } else {
  232.                         $objWriter->writeAttribute('width',         $colDimension->getWidth());
  233.                         $objWriter->writeAttribute('customWidth',    '1');
  234.                     }
  235.                     $objWriter->endElement();
  236.                 }
  237.                 
  238.             $objWriter->endElement();
  239.         } else {
  240.             throw new Exception("Invalid parameters passed.");
  241.         }

  270.     private function _writeConditionalFormatting($objWriter = null, $pSheet = null)
  271.     {
  272.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet) {
  273.             // Conditional id
  274.             $id = 1;
  275.             
  276.             // Loop trough styles in the current worksheet
  277.             foreach ($pSheet->getStyles() as $cellCoodrinate => $style) {
  278.                 if (count($style->getConditionalStyles()) > 0) {
  279.                     foreach ($style->getConditionalStyles() as $conditional) {
  280.                         
  281.                         if ($conditional->getConditionType() != PHPExcel_Style_Conditional::CONDITION_NONE) {
  282.                             // conditionalFormatting
  283.                             $objWriter->startElement('conditionalFormatting');
  284.                             $objWriter->writeAttribute('sqref',     $cellCoodrinate);
  285.                             
  286.                                 // cfRule
  287.                                 $objWriter->startElement('cfRule');
  288.                                 $objWriter->writeAttribute('type',         $conditional->getConditionType());
  289.                                 $objWriter->writeAttribute('dxfId',     $this->getParentWriter()->getStylesConditionalHashTable()->getIndexForHashCode( $conditional->getHashCode() ));
  290.                                 $objWriter->writeAttribute('priority',     $id++);
  291.                                 
  292.                                 if ($conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CELLIS
  293.                                     && $conditional->getOperatorType() != PHPExcel_Style_Conditional::OPERATOR_NONE) {
  294.                                     $objWriter->writeAttribute('operator',     $conditional->getOperatorType());
  295.                                     
  296.                                     // Formula
  297.                                     $objWriter->writeElement('formula',    $conditional->getCondition());        
  298.                                 }        
  299.                                 
  300.                                 $objWriter->endElement();
  301.                             
  302.                             $objWriter->endElement();
  303.                         }
  304.  
  305.                     }
  306.                 }
  307.             }
  308.         } else {
  309.             throw new Exception("Invalid parameters passed.");
  310.         }
  311.     }

  402.     /**

  407.      * @throws     Exception

  409.     private function _writeBreaks($objWriter = null, $pSheet = null)
  410.     {
  411.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet) {
  412.             // Get row and column breaks
  413.             $aRowBreaks = array();
  414.             $aColumnBreaks = array();
  415.             foreach ($pSheet->getBreaks() as $cell => $breakType) {
  416.                 if ($breakType == PHPExcel_Worksheet::BREAK_ROW) {
  417.                     array_push($aRowBreaks, $cell);
  418.                 } else if ($breakType == PHPExcel_Worksheet::BREAK_COLUMN) {
  419.                     array_push($aColumnBreaks, $cell);
  420.                 }
  421.             }
  422.             
  423.             // rowBreaks
  424.             if (count($aRowBreaks) > 0) {
  425.                 $objWriter->startElement('rowBreaks');
  426.                 $objWriter->writeAttribute('count',             count($aRowBreaks));
  427.                 $objWriter->writeAttribute('manualBreakCount',     count($aRowBreaks));
  428.                 
  429.                     foreach ($aRowBreaks as $cell) {
  430.                         $coords = PHPExcel_Cell::coordinateFromString($cell);
  431.                         
  432.                         $objWriter->startElement('brk');
  433.                         $objWriter->writeAttribute('id',     $coords[1]);
  434.                         $objWriter->writeAttribute('man',     '1');
  435.                         $objWriter->endElement();
  436.                     }
  437.                     
  438.                 $objWriter->endElement();
  439.             }
  440.             
  441.             // Second, write column breaks
  442.             if (count($aColumnBreaks) > 0) {
  443.                 $objWriter->startElement('colBreaks');
  444.                 $objWriter->writeAttribute('count',             count($aColumnBreaks));
  445.                 $objWriter->writeAttribute('manualBreakCount',     count($aColumnBreaks));
  446.                 
  447.                     foreach ($aColumnBreaks as $cell) {

  449.                         
  450.                         $objWriter->startElement('brk');
  451.                         $objWriter->writeAttribute('id',     PHPExcel_Cell::columnIndexFromString($coords[0]) - 1);
  452.                         $objWriter->writeAttribute('man',     '1');

  454.                     }

  456.                 $objWriter->endElement();

  463.     /**
  464.      * Write SheetData
  465.      *
  466.      * @param     xmlWriter                 $objWriter         XML Writer
  467.      * @param     PHPExcel_Worksheet    $pSheet            Worksheet
  468.      * @param     string[]                $pStringTable    String table
  469.      * @throws     Exception
  470.      */
  471.     private function _writeSheetData($objWriter = null, $pSheet = null, $pStringTable = null)
  472.     {
  473.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet && is_array($pStringTable)) {
  474.             // Flipped stringtable, for faster index searching
  475.             $aFlippedStringTable = array_flip($pStringTable);
  476.             
  477.             // sheetData
  478.             $objWriter->startElement('sheetData');
  479.  
  480.                 // Get column count
  481.                 $colCount = PHPExcel_Cell::columnIndexFromString($pSheet->getHighestColumn());
  482.             
  483.                 // Loop trough cells
  484.                 $currentRow = -1;
  485.                 $cellCollection = $pSheet->getCellCollection();
  486.                 foreach ($cellCollection as $cell) {                    
  487.                     if ($currentRow != $cell->getRow()) {
  488.                         // End previous row?
  489.                         if ($currentRow != -1) {
  490.                             $objWriter->endElement();
  491.                         }
  492.  
  493.                         // Set current row
  494.                         $currentRow = $cell->getRow();
  495.  
  496.                         // Get row dimension
  497.                         $rowDimension = $pSheet->getRowDimension($currentRow);
  498.             
  499.                         // Start a new row
  500.                         $objWriter->startElement('row');
  501.                         $objWriter->writeAttribute('r',     $currentRow);
  502.                         $objWriter->writeAttribute('spans',    '1:' . $colCount);
  503.                                     
  504.                         // Row dimensions
  505.                         if ($rowDimension->getRowHeight() >= 0) {
  506.                             $objWriter->writeAttribute('customHeight',     '1');
  507.                             $objWriter->writeAttribute('ht',             $rowDimension->getRowHeight());
  508.                         }
  509.                     }
  510.                     
  511.                     // Write cell
  512.                     $this->_writeCell($objWriter, $pSheet, $cell, $pStringTable, $aFlippedStringTable);
  513.                 }
  514.  
  515.                 // End last row?
  516.                 if ($currentRow != -1) {
  517.                     $objWriter->endElement();
  518.                 }
  519.                     
  520.             $objWriter->endElement();
  521.         } else {
  522.             throw new Exception("Invalid parameters passed.");
  523.         }
  524.     }

  526.     /**
  527.      * Write Cell
  528.      *
  529.      * @param     xmlWriter                 $objWriter                 XML Writer
  530.      * @param     PHPExcel_Worksheet    $pSheet                    Worksheet
  531.      * @param     PHPExcel_Cell        $pCell                    Cell
  532.      * @param     string[]                $pStringTable            String table
  533.      * @param     string[]                $pFlippedStringTable    String table (flipped), for faster index searching
  534.      * @throws     Exception
  535.      */
  536.     private function _writeCell($objWriter = null, $pSheet = null, $pCell = null, $pStringTable = null, $pFlippedStringTable = null)
  537.     {
  538.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet && $pCell instanceof PHPExcel_Cell && is_array($pStringTable) && is_array($pFlippedStringTable)) {
  539.             // Cell
  540.             $objWriter->startElement('c');
  541.             $objWriter->writeAttribute('r', $pCell->getCoordinate());
  542.             
  543.             // Sheet styles
  544.             $aStyles = $pSheet->getStyles();
  545.             if (isset($aStyles[$pCell->getCoordinate()])) {
  546.                 // Fix for work item 836 - http://www.codeplex.com/PHPExcel/WorkItem/View.aspx?WorkItemId=836
  547.                 // Seems the style can be empty after files pass Excel2007_Reader...
  548.                 if ($this->getParentWriter()->getStylesHashTable()->getIndexForHashCode( $aStyles[$pCell->getCoordinate()]->getHashCode() ) !== false) {
  549.                     $objWriter->writeAttribute('s', $this->getParentWriter()->getStylesHashTable()->getIndexForHashCode( $aStyles[$pCell->getCoordinate()]->getHashCode() ));
  550.                 }
  551.             }
  552.             
  553.             // If cell value is supplied, write cell value
  554.             if ($pCell->getValue() != '') {
  555.                 // Map type
  556.                 $mappedType = $pCell->getDataType();
  557.                                 
  558.                 // Write data type depending on its type
  559.                 switch (strtolower($mappedType)) {
  560.                     case 's':     // String
  561.                         $objWriter->writeAttribute('t', $mappedType);
  562.                         break;
  563.                     case 'b':     // Boolean
  564.                         $objWriter->writeAttribute('t', $mappedType);
  565.                         break;
  566.                     case 'f':     // Formula
  567.                         $objWriter->writeAttribute('t', 'str');
  568.                         break;
  569.                 }
  570.                                     
  571.                 // Write data depending on its type
  572.                 switch (strtolower($mappedType)) {
  573.                     case 's':     // String
  574.                         $objWriter->writeElement('v', $pFlippedStringTable[$pCell->getValue()]); //array_search($pCell->getValue(), $pStringTable));
  575.                         break;
  576.                     case 'f':     // Formula
  577.                         $objWriter->writeElement('f', substr($pCell->getValue(), 1));
  578.                         break;
  579.                     case 'n':     // Numeric
  580.                         $objWriter->writeElement('v', $pCell->getValue());
  581.                         break;
  582.                     case 'b':     // Boolean
  583.                         $objWriter->writeElement('v', ($pCell->getValue() ? '1' : '0'));
  584.                         break;
  585.                 }
  586.             }
  587.  
  588.             $objWriter->endElement();
  589.         } else {
  590.             throw new Exception("Invalid parameters passed.");
  591.         }
  592.     }
