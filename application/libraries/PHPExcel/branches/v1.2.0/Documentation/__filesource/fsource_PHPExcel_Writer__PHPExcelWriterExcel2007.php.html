  245.             }

  246.

  247.             // Create styles dictionaries

  248.             $this->_stylesHashTable->addFromSource(             $this->getWriterPart('Style')->allStyles($this->_spreadSheet)             );

  253.             $this->_numFmtHashTable->addFromSource(             $this->getWriterPart('Style')->allNumberFormats($this->_spreadSheet)     );
    254. 

  254.  
    255. 

  255.             // Create drawing dictionary

  256.             $this->_drawingHashTable->addFromSource(             $this->getWriterPart('Drawing')->allDrawings($this->_spreadSheet)         );

  264.             }
    265. 

  265.             
    266. 

  266.             // Add [Content_Types].xml to ZIP file

  268.

    272.

    273.

  275.             $objZip->addFromString('docProps/core.xml',             $this->getWriterPart('DocProps')->writeDocPropsCore($this->_spreadSheet));
    276. 

  276.                     
    277. 

  277.             // Add theme to ZIP file
    278. 

  278.             $objZip->addFromString('xl/theme/theme1.xml',             $this->getWriterPart('Theme')->writeTheme($this->_spreadSheet));
    279. 

  279.  
    280. 

  280.             // Add string table to ZIP file

  282.             
    283. 

  283.             // Add styles to ZIP file

  284.             $objZip->addFromString('xl/styles.xml',                 $this->getWriterPart('Style')->writeStyles($this->_spreadSheet));

    286. 

  286.             // Add workbook to ZIP file

  288.  
    289. 

  289.             // Add worksheets
    290. 

  290.             for ($i = 0; $i < $this->_spreadSheet->getSheetCount(); $i++) {

    292. 

  292.             }
    293. 

  293.  
    294. 

  294.             // Add worksheet relationships (drawings, ...)
    295. 

  295.             for ($i = 0; $i < $this->_spreadSheet->getSheetCount(); $i++) {
    296. 

  296.                 // Worksheet relationships

  297.                 $objZip->addFromString('xl/worksheets/_rels/sheet' . ($i + 1) . '.xml.rels',     $this->getWriterPart('Rels')->writeWorksheetRelationships($this->_spreadSheet->getSheet($i), ($i + 1)));
    298. 

  298.

    302.
