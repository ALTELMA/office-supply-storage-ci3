    426. 

  426.     private function _writeBorder($objWriter = null, $pBorders = null)
    427. 

  427.     {
    428. 

  428.         if ($objWriter instanceof xmlWriter && $pBorders instanceof PHPExcel_Style_Borders) {
    429. 

  429.             // Write border
    430. 

  430.             $objWriter->startElement('border');
    431. 

  431.                 // Diagonal?
    432. 

  432.                 switch ($pBorders->getDiagonalDirection()) {
    433. 

  433.                     case PHPExcel_Style_Borders::DIAGONAL_UP:
    434. 

  434.                         $objWriter->writeAttribute('diagonalUp',     'true');
    435. 

  435.                         $objWriter->writeAttribute('diagonalDown',     'false');
    436. 

  436.                         break;
    437.
