  60.     {
    61. 

  61.         if (!is_null($pSheet)) {

  62.             // Create XML writer

  72.             $objWriter->writeAttribute('xmlns', 'http://schemas.openxmlformats.org/spreadsheetml/2006/main');
    73. 

  73.             $objWriter->writeAttribute('xmlns:r', 'http://schemas.openxmlformats.org/officeDocument/2006/relationships');
    74. 

  74.             
    75. 

  75.                 // sheetPr
    76. 

  76.                 $this->_writeSheetPr($objWriter, $pSheet);
    77. 

  77.             
    78. 

  78.                 // Dimension
    79. 

  79.                 $this->_writeDimension($objWriter, $pSheet);
    80. 

  80.                 
    81. 

  81.                 // sheetViews
    82. 

  82.                 $this->_writeSheetViews($objWriter);
    83. 

  83.             
    84. 

  84.                 // sheetFormatPr
    85. 

  85.                 $this->_writeSheetFormatPr($objWriter);
    86. 

  86.                 
    87. 

  87.                 // cols
    88. 

  88.                 $this->_writeCols($objWriter, $pSheet);
    89. 

  89.                         
    90. 

  90.                 // sheetData
    91. 

  91.                 $this->_writeSheetData($objWriter, $pSheet, $pStringTable);
    92. 

  92.             
    93. 

  93.                 // sheetProtection
    94. 

  94.                 $this->_writeSheetProtection($objWriter, $pSheet);
    95. 

  95.       
    96. 

  96.                 // protectedRanges

  97.                 $this->_writeProtectedRanges($objWriter, $pSheet);
    98. 

  98.                 
    99. 

  99.                 // autoFilter
    100. 

  100.                 $this->_writeAutoFilter($objWriter, $pSheet);
    101. 

  101.                 
    102. 

  102.                 // mergeCells
    103. 

  103.                 $this->_writeMergeCells($objWriter, $pSheet);
    104. 

  104.                 
    105. 

  105.                 // conditionalFormatting

  106.                 $this->_writeConditionalFormatting($objWriter, $pSheet);

    108.

  110.                                 
    111.

  112.                 $this->_writePageMargins($objWriter, $pSheet);
    113. 

  113.  
    114. 

  114.                 // Page setup
    115. 

  115.                 $this->_writePageSetup($objWriter, $pSheet);

  118.                 $this->_writeHeaderFooter($objWriter, $pSheet);

    120.

  121.                 $this->_writeBreaks($objWriter, $pSheet);

    126. 

  126.             $objWriter->endElement();

  225.      */
    226. 

  226.     private function _writeCols($objWriter = null, $pSheet = null)
    227. 

  227.     {
    228. 

  228.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet) {
    229. 

  229.             // cols
    230. 

  230.             $objWriter->startElement('cols');
    231. 

  231.                     
    232. 

  232.                 // Check if there is at least one column dimension specified. If not, create one.
    233. 

  233.                 if (count($pSheet->getColumnDimensions()) == 0) {
    234. 

  234.                     $pSheet->getColumnDimension('A')->setWidth(10);
    235. 

  235.                 }
    236. 

  236.                     
    237. 

  237.                 // Loop trough column dimensions
    238. 

  238.                 foreach ($pSheet->getColumnDimensions() as $colDimension) {
    239. 

  239.                     // col
    240. 

  240.                     $objWriter->startElement('col');
    241. 

  241.                     $objWriter->writeAttribute('min',     PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
    242. 

  242.                     $objWriter->writeAttribute('max',     PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
    243. 

  243.                         
    244. 

  244.                     if ($colDimension->getWidth() < 0) {
    245. 

  245.                         $objWriter->writeAttribute('bestFit',         '1');
    246. 

  246.                         $objWriter->writeAttribute('width',         '10');
    247. 

  247.                     } else {
    248. 

  248.                         $objWriter->writeAttribute('width',         $colDimension->getWidth());
    249. 

  249.                         $objWriter->writeAttribute('customWidth',    '1');
    250. 

  250.                     }
    251. 

  251.                     $objWriter->endElement();
    252. 

  252.                 }
    253. 

  253.                 
    254. 

  254.             $objWriter->endElement();
    255. 

  255.         } else {
    256. 

  256.             throw new Exception("Invalid parameters passed.");
    257. 

  257.         }
    258. 

  258.     }

  307.     {

  308.         if ($objWriter instanceof xmlWriter && $pSheet instanceof PHPExcel_Worksheet) {

  309.             // Conditional id

  310.             $id = 1;
    311. 

  311.             
    312. 

  312.             // Loop trough styles in the current worksheet
    313. 

  313.             foreach ($pSheet->getStyles() as $cellCoodrinate => $style) {
    314. 

  314.                 if (count($style->getConditionalStyles()) > 0) {
    315. 

  315.                     foreach ($style->getConditionalStyles() as $conditional) {
    316. 

  316.                         
    317. 

  317.                         if ($conditional->getConditionType() != PHPExcel_Style_Conditional::CONDITION_NONE) {
    318. 

  318.                             // conditionalFormatting
    319. 

  319.                             $objWriter->startElement('conditionalFormatting');
    320. 

  320.                             $objWriter->writeAttribute('sqref',     $cellCoodrinate);

  321.                             
    322.

    323. 

  323.                                 $objWriter->startElement('cfRule');
    324. 

  324.                                 $objWriter->writeAttribute('type',         $conditional->getConditionType());
    325. 

  325.                                 $objWriter->writeAttribute('dxfId',     $this->getParentWriter()->getStylesConditionalHashTable()->getIndexForHashCode( $conditional->getHashCode() ));
    326. 

  326.                                 $objWriter->writeAttribute('priority',     $id++);
    327. 

  327.                                 
    328. 

  328.                                 if ($conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CELLIS
    329. 

  329.                                     && $conditional->getOperatorType() != PHPExcel_Style_Conditional::OPERATOR_NONE) {
    330. 

  330.                                     $objWriter->writeAttribute('operator',     $conditional->getOperatorType());
    331. 

  331.                                     
    332. 

  332.                                     // Formula
    333. 

  333.                                     $objWriter->writeElement('formula',    $conditional->getCondition());        
    334. 

  334.                                 }        
    335. 

  335.                                 
    336. 

  336.                                 $objWriter->endElement();
    337. 

  337.                             
    338. 

  338.                             $objWriter->endElement();
    339. 

  339.                         }
    340. 

  340.  
    341. 

  341.                     }
    342. 

  342.                 }
