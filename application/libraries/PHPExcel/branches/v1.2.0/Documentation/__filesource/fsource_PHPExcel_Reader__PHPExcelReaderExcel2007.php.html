  90.                     foreach ($xmlStrings->si as $val) {

  91.                         $sharedStrings[] = (string) $val->t;

  92.                     }
    93. 

  93.                     
    94. 

  94.                     $worksheets = array();
    95. 

  95.                     foreach ($relsWorkbook->Relationship as $ele) {
    96. 

  96.                         if ($ele["Type"] == "http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet") {
    97. 

  97.                             $worksheets[(string) $ele["Id"]] = $ele["Target"];
    98. 

  98.                         }
    99. 

  99.                     }
    100. 

  100.                     
    101. 

  101.                     $styles = array();
    102. 

  102.                     $xpath = self::array_item($relsWorkbook->xpath("rel:Relationship[@Type='http://schemas.openxmlformats.org/officeDocument/2006/relationships/styles']"));
    103. 

  103.                     $xmlStyles = simplexml_load_file("zip://$pFilename#$dir/$xpath[Target]"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
    104. 

  104.                     $numFmts = $xmlStyles->numFmts[0];
    105. 

  105.                     $numFmts->registerXPathNamespace("sml", "http://schemas.openxmlformats.org/spreadsheetml/2006/main");
    106. 

  106.                     foreach ($xmlStyles->cellStyles->cellStyle as $style) {
    107. 

  107.                         $xf = $xmlStyles->cellXfs->xf[intval($style["xfId"])];
    108. 

  108.                         $styles[] = (object) array(
    109. 

  109.                             "numFmt" => self::array_item($numFmts->xpath("sml:numFmt[@numFmtId=$xf[numFmtId]]")),
    110. 

  110.                             "font" => $xmlStyles->fonts->font[intval($xf["fontId"])],
    111. 

  111.                             "fill" => $xmlStyles->fills->fill[intval($xf["fillId"])],
    112. 

  112.                             "border" => $xmlStyles->borders->border[intval($xf["borderId"])],
    113. 

  113.                             "alignment" => $xf->alignment,
    114. 

  114.                         );
    115. 

  115.                     }
    116. 

  116.                     
    117. 

  117.                     $dxfs = array();
    118. 

  118.                     foreach ($xmlStyles->dxfs->dxf as $dxf) {
    119. 

  119.                         $style = new PHPExcel_Style;
    120. 

  120.                         $this->_readStyle($style, $dxf);
    121. 

  121.                         $dxfs[] = $style;
    122. 

  122.                     }
    123. 

  123.                     
    124. 

  124.                     $xmlWorkbook = simplexml_load_file("zip://$pFilename#$rel[Target]"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
    125. 

  125.                     foreach ($xmlWorkbook->sheets->sheet as $eleSheet) {
    126. 

  126.                         $docSheet = $excel->createSheet();
    127. 

  127.                         $docSheet->setTitle((string) $eleSheet["name"]);
    128. 

  128.                         $fileWorkheet = $worksheets[(string) self::array_item($eleSheet->attributes("http://schemas.openxmlformats.org/officeDocument/2006/relationships"), "id")];
    129. 

  129.                         $xmlSheet = simplexml_load_file("zip://$pFilename#$dir/$fileWorkheet"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
    130. 

  130.                         foreach ($xmlSheet->cols->col as $col) {
    131. 

  131.                             for ($i=intval($col["min"])-1; $i < intval($col["max"]); $i++) {
    132. 

  132.                                 $docSheet->getColumnDimension(PHPExcel_Cell::stringFromColumnIndex($i))->setWidth($col["bestFit"] ? -1 : floatval($col["width"]));
    133. 

  133.                             }
    134. 

  134.                         }
    135. 

  135.                         
    136. 

  136.                         foreach ($xmlSheet->sheetData->row as $row) {
    137. 

  137.                             if ($row["ht"]) {
    138. 

  138.                                 $docSheet->getRowDimension(intval($row["r"]))->setRowHeight(floatval($row["ht"]));
    139. 

  139.                             }
    140. 

  140.                             foreach ($row->c as $c) {
    141. 

  141.                                 $r = (string) $c["r"];
    142. 

  142.                                 switch ($c["t"]) {
    143. 

  143.                                     case "s": $value = $sharedStrings[intval($c->v)]; break;
    144. 

  144.                                     case "b": $value = (bool) $c->v; break;
    145. 

  145.                                     case "str": $value = "=$c->f"; break;
    146. 

  146.                                     case "inlineStr": $value = (string) $c->is->t; break;
    147. 

  147.                                     default: $value = (string) $c->v; break;
    148. 

  148.                                 }
    149. 

  149.                                 if ($value) {
    150. 

  150.                                     $docSheet->setCellValue($r, $value);
    151. 

  151.                                 }
    152. 

  152.                                 if ($c["s"]) {
    153. 

  153.                                     $this->_readStyle($docSheet->getStyle($r), $styles[intval($c["s"])]);
    154. 

  154.                                 }
    155. 

  155.                             }
    156. 

  156.                         }
    157. 

  157.                         
    158. 

  158.                         $conditionals = array();
    159. 

  159.                         foreach ($xmlSheet->conditionalFormatting as $conditional) {
    160. 

  160.                             foreach ($conditional->cfRule as $cfRule) {
    161. 

  161.                                 $conditionals[(string) $conditional["sqref"]][intval($cfRule["priority"])] = $cfRule;
    162. 

  162.                             }
    163. 

  163.                         }
    164. 

  164.                         foreach ($conditionals as $ref => $cfRules) {
    165. 

  165.                             ksort($cfRules);
    166. 

  166.                             $conditionalStyles = array();
    167. 

  167.                             foreach ($cfRules as $cfRule) {
    168. 

  168.                                 $objConditional = new PHPExcel_Style_Conditional;
    169. 

  169.                                 $objConditional->setConditionType((string) $cfRule["type"]);
    170. 

  170.                                 $objConditional->setOperatorType((string) $cfRule["operator"]);
    171. 

  171.                                 $objConditional->setCondition((string) $cfRule->formula);
    172. 

  172.                                 $objConditional->setStyle($dxfs[intval($cfRule["dxfId"])]);
    173. 

  173.                                 $conditionalStyles[] = $objConditional;
    174. 

  174.                             }
    175. 

  175.                             $docSheet->getStyle($ref)->setConditionalStyles($conditionalStyles);
    176. 

  176.                         }
    177. 

  177.                         
    178. 

  178.                         $docPageMargins = $docSheet->getPageMargins();
    179. 

  179.                         $docPageMargins->setLeft(floatval($xmlSheet->pageMargins["left"]));
    180. 

  180.                         $docPageMargins->setRight(floatval($xmlSheet->pageMargins["right"]));
    181. 

  181.                         $docPageMargins->setTop(floatval($xmlSheet->pageMargins["top"]));
    182. 

  182.                         $docPageMargins->setBottom(floatval($xmlSheet->pageMargins["bottom"]));
    183. 

  183.                         $docPageMargins->setHeader(floatval($xmlSheet->pageMargins["header"]));
    184. 

  184.                         $docPageMargins->setFooter(floatval($xmlSheet->pageMargins["footer"]));
    185. 

  185.                         
    186. 

  186.                         $docPageSetup = $docSheet->getPageSetup();
    187. 

  187.                         $docPageSetup->setOrientation((string) $xmlSheet->pageSetup["orientation"]);
    188. 

  188.                         $docPageSetup->setPaperSize(intval($xmlSheet->pageSetup["paperSize"]));
    189. 

  189.                         
    190. 

  190.                         $docHeaderFooter = $docSheet->getHeaderFooter();
    191. 

  191.                         $docHeaderFooter->setDifferentOddEven($xmlSheet->headerFooter["differentOddEven"] == 'true');
    192. 

  192.                         $docHeaderFooter->setDifferentFirst($xmlSheet->headerFooter["differentFirst"] == 'true');
    193. 

  193.                         $docHeaderFooter->setScaleWithDocument($xmlSheet->headerFooter["scaleWithDoc"] == 'true');
    194. 

  194.                         $docHeaderFooter->setAlignWithMargins($xmlSheet->headerFooter["alignWithMargins"] == 'true');
    195. 

  195.                         $docHeaderFooter->setOddHeader((string) $xmlSheet->headerFooter->oddHeader);
    196. 

  196.                         $docHeaderFooter->setOddFooter((string) $xmlSheet->headerFooter->oddFooter);
    197. 

  197.                         $docHeaderFooter->setEvenHeader((string) $xmlSheet->headerFooter->evenHeader);
    198. 

  198.                         $docHeaderFooter->setEvenFooter((string) $xmlSheet->headerFooter->evenFooter);
    199. 

  199.                         $docHeaderFooter->setFirstHeader((string) $xmlSheet->headerFooter->firstHeader);
    200. 

  200.                         $docHeaderFooter->setFirstFooter((string) $xmlSheet->headerFooter->firstFooter);
    201. 

  201.                         
    202. 

  202.                         /* //! drawing
    203. 

  203. $objDrawing = new PHPExcel_Worksheet_Drawing();
    204. 

  204. $objDrawing->setName('Paid');
    205. 

  205. $objDrawing->setDescription('Paid');
    206. 

  206. $objDrawing->setPath('./images/paid.png');
    207. 

  207. $objDrawing->setCoordinates('B15');
    208. 

  208. $objDrawing->setOffsetX(110);
    209. 

  209. $objDrawing->setRotation(25);

    211. 

  211.                         */
    212. 

  212.                     }
    213. 

  213.                     
    214. 

  214.                     $excel->setActiveSheetIndex(intval($xmlWorkbook->bookView->workbookView["activeTab"]));
    215. 

  215.                 break;
    216. 

  216.             }
    217. 

  217.         }
    218. 

  218.

  222.     private function _readStyle($docStyle, $style) {
    223. 

  223.         // format code
    224. 

  224.         $docStyle->getNumberFormat()->setFormatCode((string) $style->numFmt["formatCode"]);
    225. 

  225.         
    226. 

  226.         // font
    227. 

  227.         $docStyle->getFont()->setName((string) $style->font->name["val"]);
    228. 

  228.         $docStyle->getFont()->setSize((string) $style->font->sz["val"]);
    229. 

  229.         $docStyle->getFont()->setBold($style->font->b["val"] == 'true');
    230. 

  230.         $docStyle->getFont()->setItalic($style->font->i["val"] == 'true');
    231. 

  231.         $docStyle->getFont()->setStriketrough($style->font->strike["val"] == 'true');
    232. 

  232.         $docStyle->getFont()->getColor()->setARGB((string) $style->font->color["rgb"]);
    233. 

  233.         $docStyle->getFont()->setUnderline((string) $style->font->u["val"]);
    234. 

  234.         
    235. 

  235.         // fill
    236. 

  236.         if ($style->fill->gradientFill) {
    237. 

  237.             $gradientFill = $style->fill->gradientFill[0];
    238. 

  238.             $docStyle->getFill()->setFillType((string) $gradientFill["type"]);
    239. 

  239.             $docStyle->getFill()->setRotation(floatval($gradientFill["degree"]));
    240. 

  240.             $gradientFill->registerXPathNamespace("sml", "http://schemas.openxmlformats.org/spreadsheetml/2006/main");
    241. 

  241.             $docStyle->getFill()->getStartColor()->setARGB((string) self::array_item($gradientFill->xpath("sml:stop[@position=0]"))->color["rgb"]);
    242. 

  242.             $docStyle->getFill()->getEndColor()->setARGB((string) self::array_item($gradientFill->xpath("sml:stop[@position=1]"))->color["rgb"]);
    243. 

  243.         } elseif ($style->fill->patternFill) {
    244. 

  244.             $docStyle->getFill()->setFillType((string) $style->fill->patternFill["patternType"]);
    245. 

  245.             if ($style->fill->patternFill->fgColor) {
    246. 

  246.                 $docStyle->getFill()->getStartColor()->setARGB((string) $style->fill->patternFill->fgColor["rgb"]);
    247. 

  247.             }
    248. 

  248.             if ($style->fill->patternFill->bgColor) {
    249. 

  249.                 $docStyle->getFill()->getEndColor()->setARGB((string) $style->fill->patternFill->bgColor["rgb"]);
    250. 

  250.             }
    251. 

  251.         }
    252. 

  252.         
    253. 

  253.         // border
    254. 

  254.         if ($style->border["diagonalUp"] == 'true') {
    255. 

  255.             $docStyle->getBorders()->setDiagonalDirection(PHPExcel_Style_Borders::DIAGONAL_UP);
    256. 

  256.         } elseif ($style->border["diagonalDown"] == 'true') {
    257. 

  257.             $docStyle->getBorders()->setDiagonalDirection(PHPExcel_Style_Borders::DIAGONAL_DOWN);
    258. 

  258.         }
    259. 

  259.         $docStyle->getBorders()->setOutline($style->border["outline"] == 'true');
    260. 

  260.         $this->_readBorder($docStyle->getBorders()->getLeft(), $style->border->left);
    261. 

  261.         $this->_readBorder($docStyle->getBorders()->getRight(), $style->border->right);
    262. 

  262.         $this->_readBorder($docStyle->getBorders()->getTop(), $style->border->top);
    263. 

  263.         $this->_readBorder($docStyle->getBorders()->getBottom(), $style->border->bottom);
    264. 

  264.         $this->_readBorder($docStyle->getBorders()->getDiagonal(), $style->border->diagonal);
    265. 

  265.         $this->_readBorder($docStyle->getBorders()->getVertical(), $style->border->vertical);
    266. 

  266.         $this->_readBorder($docStyle->getBorders()->getHorizontal(), $style->border->horizontal);
    267. 

  267.         
    268. 

  268.         // alignment
    269. 

  269.         $docStyle->getAlignment()->setHorizontal((string) $style->alignment["horizontal"]);
    270. 

  270.         $docStyle->getAlignment()->setVertical((string) $style->alignment["vertical"]);
    271. 

  271.         $docStyle->getAlignment()->setTextRotation(intval($style->alignment["textRotation"]));
    272. 

  272.     }

  274.     private function _readBorder($docBorder, $eleBorder) {
    275. 

  275.         if (isset($eleBorder["style"])) {
    276. 

  276.             $docBorder->setBorderStyle((string) $eleBorder["style"]);
    277. 

  277.         }
    278. 

  278.         if (isset($eleBorder->color["rgb"])) {
    279. 

  279.             $docBorder->getColor()->setARGB((string) $eleBorder->color["rgb"]);
    280. 

  280.         }
    281. 

  281.     }
