  51.         $rels = simplexml_load_file("zip://$pFilename#_rels/.rels"); //~ http://schemas.openxmlformats.org/package/2006/relationships");

  54.                 case "http://schemas.openxmlformats.org/package/2006/relationships/metadata/core-properties":

  59.                     $docProps = $excel->getProperties();

  61.                     $docProps->setLastModifiedBy((string) self::array_item($xmlCore->xpath("cp:lastModifiedBy")));
  62.                     $docProps->setCreated(strtotime(self::array_item($xmlCore->xpath("dcterms:created")))); //! respect xsi:type

  65.                     $docProps->setDescription((string) self::array_item($xmlCore->xpath("dc:description")));
  66.                     $docProps->setSubject((string) self::array_item($xmlCore->xpath("dc:subject")));
  67.                 break;
  68.                 
  69.                 case "http://schemas.openxmlformats.org/officeDocument/2006/relationships/officeDocument":
  70.                     $dir = dirname($rel["Target"]);
  71.                     $relsWorkbook = simplexml_load_file("zip://$pFilename#$dir/_rels/" . basename($rel["Target"]) . ".rels"); //~ http://schemas.openxmlformats.org/package/2006/relationships");
  72.                     $relsWorkbook->registerXPathNamespace("rel", "http://schemas.openxmlformats.org/package/2006/relationships");
  73.                     
  74.                     $sharedStrings = array();
  75.                     $xpath = self::array_item($relsWorkbook->xpath("rel:Relationship[@Type='http://schemas.openxmlformats.org/officeDocument/2006/relationships/sharedStrings']"));
  76.                     $xmlStrings = simplexml_load_file("zip://$pFilename#$dir/$xpath[Target]"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  77.                     foreach ($xmlStrings->si as $val) {
  78.                         $sharedStrings[] = (string) $val->t;
  79.                     }
  80.                     
  81.                     $worksheets = array();
  82.                     foreach ($relsWorkbook->Relationship as $ele) {
  83.                         if ($ele["Type"] == "http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet") {
  84.                             $worksheets[(string) $ele["Id"]] = $ele["Target"];
  85.                         }
  86.                     }
  87.                     
  88.                     $styles = array();
  89.                     $xpath = self::array_item($relsWorkbook->xpath("rel:Relationship[@Type='http://schemas.openxmlformats.org/officeDocument/2006/relationships/styles']"));
  90.                     $xmlStyles = simplexml_load_file("zip://$pFilename#$dir/$xpath[Target]"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  91.                     $numFmts = $xmlStyles->numFmts[0];
  92.                     $numFmts->registerXPathNamespace("sml", "http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  93.                     foreach ($xmlStyles->cellStyles->cellStyle as $style) {
  94.                         $xf = $xmlStyles->cellXfs->xf[intval($style["xfId"])];
  95.                         $styles[] = (object) array(
  96.                             "numFmt" => self::array_item($numFmts->xpath("sml:numFmt[@numFmtId=$xf[numFmtId]]")),
  97.                             "font" => $xmlStyles->fonts->font[intval($xf["fontId"])],
  98.                             "fill" => $xmlStyles->fills->fill[intval($xf["fillId"])],
  99.                             "border" => $xmlStyles->borders->border[intval($xf["borderId"])],
  100.                             "alignment" => $xf->alignment,
  101.                         );
  102.                     }
  103.                     
  104.                     $dxfs = array();
  105.                     foreach ($xmlStyles->dxfs->dxf as $dxf) {
  106.                         $style = new PHPExcel_Style;
  107.                         $this->_readStyle($style, $dxf);
  108.                         $dxfs[] = $style;
  109.                     }
  110.                     
  111.                     $xmlWorkbook = simplexml_load_file("zip://$pFilename#$rel[Target]"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  112.                     foreach ($xmlWorkbook->sheets->sheet as $eleSheet) {
  113.                         $docSheet = $excel->createSheet();
  114.                         $docSheet->setTitle((string) $eleSheet["name"]);
  115.                         $fileWorkheet = $worksheets[(string) self::array_item($eleSheet->attributes("http://schemas.openxmlformats.org/officeDocument/2006/relationships"), "id")];
  116.                         $xmlSheet = simplexml_load_file("zip://$pFilename#$dir/$fileWorkheet"); //~ http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  117.                         foreach ($xmlSheet->cols->col as $col) {
  118.                             for ($i=intval($col["min"])-1; $i < intval($col["max"]); $i++) {
  119.                                 $docSheet->getColumnDimension(PHPExcel_Cell::stringFromColumnIndex($i))->setWidth($col["bestFit"] ? -1 : floatval($col["width"]));
  120.                             }
  121.                         }
  122.                         
  123.                         foreach ($xmlSheet->sheetData->row as $row) {
  124.                             if ($row["ht"]) {
  125.                                 $docSheet->getRowDimension(intval($row["r"]))->setRowHeight(floatval($row["ht"]));
  126.                             }
  127.                             foreach ($row->c as $c) {
  128.                                 $r = (string) $c["r"];
  129.                                 switch ($c["t"]) {
  130.                                     case "s": $value = $sharedStrings[intval($c->v)]; break;
  131.                                     case "b": $value = (bool) $c->v; break;
  132.                                     case "str": $value = "=$c->f"; break;
  133.                                     case "inlineStr": $value = (string) $c->is->t; break;
  134.                                     default: $value = (string) $c->v; break;
  135.                                 }
  136.                                 if ($value) {
  137.                                     $docSheet->setCellValue($r, $value);
  138.                                 }
  139.                                 if ($c["s"]) {
  140.                                     $this->_readStyle($docSheet->getStyle($r), $styles[intval($c["s"])]);
  141.                                 }
  142.                             }
  143.                         }
  144.                         
  145.                         $conditionals = array();
  146.                         foreach ($xmlSheet->conditionalFormatting as $conditional) {
  147.                             foreach ($conditional->cfRule as $cfRule) {
  148.                                 $conditionals[(string) $conditional["sqref"]][intval($cfRule["priority"])] = $cfRule;
  149.                             }
  150.                         }
  151.                         foreach ($conditionals as $ref => $cfRules) {
  152.                             ksort($cfRules);
  153.                             $conditionalStyles = array();
  154.                             foreach ($cfRules as $cfRule) {
  155.                                 $objConditional = new PHPExcel_Style_Conditional;
  156.                                 $objConditional->setConditionType((string) $cfRule["type"]);
  157.                                 $objConditional->setOperatorType((string) $cfRule["operator"]);
  158.                                 $objConditional->setCondition((string) $cfRule->formula);
  159.                                 $objConditional->setStyle($dxfs[intval($cfRule["dxfId"])]);
  160.                                 $conditionalStyles[] = $objConditional;
  161.                             }
  162.                             $docSheet->getStyle($ref)->setConditionalStyles($conditionalStyles);
  163.                         }
  164.                         
  165.                         $docPageMargins = $docSheet->getPageMargins();
  166.                         $docPageMargins->setLeft(floatval($xmlSheet->pageMargins["left"]));
  167.                         $docPageMargins->setRight(floatval($xmlSheet->pageMargins["right"]));
  168.                         $docPageMargins->setTop(floatval($xmlSheet->pageMargins["top"]));
  169.                         $docPageMargins->setBottom(floatval($xmlSheet->pageMargins["bottom"]));
  170.                         $docPageMargins->setHeader(floatval($xmlSheet->pageMargins["header"]));
  171.                         $docPageMargins->setFooter(floatval($xmlSheet->pageMargins["footer"]));
  172.                         
  173.                         $docPageSetup = $docSheet->getPageSetup();
  174.                         $docPageSetup->setOrientation((string) $xmlSheet->pageSetup["orientation"]);
  175.                         $docPageSetup->setPaperSize(intval($xmlSheet->pageSetup["paperSize"]));
  176.                         
  177.                         $docHeaderFooter = $docSheet->getHeaderFooter();
  178.                         $docHeaderFooter->setDifferentOddEven($xmlSheet->headerFooter["differentOddEven"] == 'true');
  179.                         $docHeaderFooter->setDifferentFirst($xmlSheet->headerFooter["differentFirst"] == 'true');
  180.                         $docHeaderFooter->setScaleWithDocument($xmlSheet->headerFooter["scaleWithDoc"] == 'true');
  181.                         $docHeaderFooter->setAlignWithMargins($xmlSheet->headerFooter["alignWithMargins"] == 'true');
  182.                         $docHeaderFooter->setOddHeader((string) $xmlSheet->headerFooter->oddHeader);
  183.                         $docHeaderFooter->setOddFooter((string) $xmlSheet->headerFooter->oddFooter);
  184.                         $docHeaderFooter->setEvenHeader((string) $xmlSheet->headerFooter->evenHeader);
  185.                         $docHeaderFooter->setEvenFooter((string) $xmlSheet->headerFooter->evenFooter);
  186.                         $docHeaderFooter->setFirstHeader((string) $xmlSheet->headerFooter->firstHeader);
  187.                         $docHeaderFooter->setFirstFooter((string) $xmlSheet->headerFooter->firstFooter);
  188.                         
  189.                         /* //! drawing
  190. $objDrawing = new PHPExcel_Worksheet_Drawing();
  191. $objDrawing->setName('Paid');
  192. $objDrawing->setDescription('Paid');
  193. $objDrawing->setPath('./images/paid.png');
  194. $objDrawing->setCoordinates('B15');
  195. $objDrawing->setOffsetX(110);
  196. $objDrawing->setRotation(25);
  197. $objDrawing->setWorksheet($objPHPExcel->getActiveSheet());
  198.                         */
  199.                     }
  200.                     
  201.                     $excel->setActiveSheetIndex(intval($xmlWorkbook->bookView->workbookView["activeTab"]));
  202.                 break;
  203.             }
  204.         }
  205.         
  206.         return $excel;

  209.     private function _readStyle($docStyle, $style) {
  210.         // format code
  211.         $docStyle->getNumberFormat()->setFormatCode((string) $style->numFmt["formatCode"]);
  212.         
  213.         // font
  214.         $docStyle->getFont()->setName((string) $style->font->name["val"]);
  215.         $docStyle->getFont()->setSize((string) $style->font->sz["val"]);
  216.         $docStyle->getFont()->setBold($style->font->b["val"] == 'true');
  217.         $docStyle->getFont()->setItalic($style->font->i["val"] == 'true');
  218.         $docStyle->getFont()->setStriketrough($style->font->strike["val"] == 'true');
  219.         $docStyle->getFont()->getColor()->setARGB((string) $style->font->color["rgb"]);
  220.         $docStyle->getFont()->setUnderline((string) $style->font->u["val"]);
  221.         
  222.         // fill
  223.         if ($style->fill->gradientFill) {
  224.             $gradientFill = $style->fill->gradientFill[0];
  225.             $docStyle->getFill()->setFillType((string) $gradientFill["type"]);
  226.             $docStyle->getFill()->setRotation(floatval($gradientFill["degree"]));
  227.             $gradientFill->registerXPathNamespace("sml", "http://schemas.openxmlformats.org/spreadsheetml/2006/main");
  228.             $docStyle->getFill()->getStartColor()->setARGB((string) self::array_item($gradientFill->xpath("sml:stop[@position=0]"))->color["rgb"]);
  229.             $docStyle->getFill()->getEndColor()->setARGB((string) self::array_item($gradientFill->xpath("sml:stop[@position=1]"))->color["rgb"]);
  230.         } elseif ($style->fill->patternFill) {
  231.             $docStyle->getFill()->setFillType((string) $style->fill->patternFill["patternType"]);
  232.             $docStyle->getFill()->getStartColor()->setARGB((string) $style->fill->patternFill->fgColor["rgb"]);
  233.             $docStyle->getFill()->getEndColor()->setARGB((string) $style->fill->patternFill->bgColor["rgb"]);
  234.         }
  235.         
  236.         // border
  237.         if ($style->border["diagonalUp"] == 'true') {
  238.             $docStyle->getBorders()->setDiagonalDirection(PHPExcel_Style_Borders::DIAGONAL_UP);
  239.         } elseif ($style->border["diagonalDown"] == 'true') {
  240.             $docStyle->getBorders()->setDiagonalDirection(PHPExcel_Style_Borders::DIAGONAL_DOWN);
  241.         }
  242.         $docStyle->getBorders()->setOutline($style->border["outline"] == 'true');
  243.         $this->_readBorder($docStyle->getBorders()->getLeft(), $style->border->left);
  244.         $this->_readBorder($docStyle->getBorders()->getRight(), $style->border->right);
  245.         $this->_readBorder($docStyle->getBorders()->getTop(), $style->border->top);
  246.         $this->_readBorder($docStyle->getBorders()->getBottom(), $style->border->bottom);
  247.         $this->_readBorder($docStyle->getBorders()->getDiagonal(), $style->border->diagonal);
  248.         $this->_readBorder($docStyle->getBorders()->getVertical(), $style->border->vertical);
  249.         $this->_readBorder($docStyle->getBorders()->getHorizontal(), $style->border->horizontal);
  250.         
  251.         // alignment
  252.         $docStyle->getAlignment()->setHorizontal((string) $style->alignment["horizontal"]);
  253.         $docStyle->getAlignment()->setVertical((string) $style->alignment["vertical"]);
  254.         $docStyle->getAlignment()->setTextRotation(intval($style->alignment["textRotation"]));
  255.     }

  257.     private function _readBorder($docBorder, $eleBorder) {
  258.         if (isset($eleBorder["style"])) {
  259.             $docBorder->setBorderStyle((string) $eleBorder["style"]);
  260.         }
  261.         if (isset($eleBorder->color["rgb"])) {
  262.             $docBorder->getColor()->setARGB((string) $eleBorder->color["rgb"]);
  263.         }
  264.     }
